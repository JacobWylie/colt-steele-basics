// Use push() to add to the end of an array
let colors = ['red', 'green', 'orange'];
colors.push('yellow');
// = ['red', 'green', 'orange', 'yellow']
// Also returns length of array after new item added


//Use pop() to remove the last item in an array
let colors1 = ['red', 'orange', 'yellow'];
colors1.pop();
// = ['red', 'orange']
// Also returns removed element


// Use unshift() to add to the front of an array
let colors2 = ['red', 'orange', 'yellow'];
colors2.unshift('blue');
// = ['blue', 'red', 'orange', 'yellow']


// Use shift() to remove the first item in an array
let colors3 = ['red', 'orange', 'yellow'];
colors3.shift('red');
// = ['orange', 'yellow']
// also returns removed element


// use indexOf() to find the index of an item in an array
let friends = ['Charlie', 'Liz', 'David', 'Mattias', 'Liz'];
//returns the first index at which a given element can be found
friends.indexOf('David'); // 2
friends.indexOf('Liz'); // 1
// returns -1 if the element is not present
friends.indexOf('Hagrid'); // -1


// use slice() to copy parts of an array
let fruits = ['banana', 'orange', 'lemon', 'apple', 'mango'];
// use slice() to copy the 2nd and 3rd fruits
// specify index where the new array starts(1) and ends(2)
let citrus = fruits.slice(1, 3);
// this does not alter the original fruits array
// citrus contains ['orange', 'lemon']
// fruits contains ['banana', 'orange', 'lemon', 'apple', 'mango']

//you can also use slice() to copy an entire array
let nums = [1, 2, 3];
let otherNums = nums.slice();
// not arrays are = [1, 2, 3]
